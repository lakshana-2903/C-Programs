#include <stdio.h>
int main(){
	int n;
	scanf("%d", &n);
	int arr[n];
	bool soln = true;
	for(int i=0;i<n;i++){
		scanf("%d", &arr[i]);
	}
	for(int i=0;i<n-1;i++){
		if(arr[i]<arr[i+1]){ 
			continue;
		}
		else{
			soln = false;
		}
	}
	if(soln){
		printf("SORTED ARRAY\n");	
	}
	else{
		printf("NOT AN SORTED ARRAY\n");	
	}
	return 0;
}
