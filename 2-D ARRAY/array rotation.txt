#include <stdio.h>
int  main(){
	int n;
	scanf("%d", &n);
	int arr[n];
	for(int i=0;i<n;i++){
		scanf("%d", &arr[i]);
	}
	int k;
	scanf("%d", &k);
	if(k>=n){
		printf("INVALID K");
		return 0;	
	}
	while(k>0){
		int temp = arr[n-1];
		for(int i=n-1;i>0;i--){
			arr[i] = arr[i-1];
		}
		arr[0] = temp;	
		k--;
	}
	for(int j=0;j<n;j++){
		printf("%d ", arr[j]);
	}
	return 0;
}
