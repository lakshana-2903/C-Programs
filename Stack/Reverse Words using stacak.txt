#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int main(){
    int top = -1;
    char s[100];
    scanf("%[^\n]", s);
    char stack[100][100];
    for(int i=0;i<strlen(s);i++){
        char new[100];
        int j=0;
        while(s[i]!=' ' && s[i]!='\0'){
            new[j] = s[i];
            j++;
            i++;
        }
        new[j] = '\0';
        strcpy(stack[++top], new);
    }
    while(top!=-1){
        printf("%s ", stack[top--]);
    }
    return 0;
}
