#include<stdio.h>
#include<stdlib.h>
#include<string.h>
int main(){
    char s[100];
    int top = -1;
    char stack[100];
    scanf("%[^\n]", s);
    int n = strlen(s);
    for(int i=0;i<n;i++){
        if(s[i]=='*' && top==-1){
            continue;
        }
        if(s[i]!='*'){
            stack[++top] = s[i];
            continue;
        }
        top--;
    }
    if(top==-1){
        printf("EMPTY!");
        return 0;
    }
    for(int i=0;i<=top;i++){
        printf("%c", stack[i]);
    }
}
