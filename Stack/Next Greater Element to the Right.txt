#include<stdio.h>
#include<stdlib.h>
int main(){
    int n;
    scanf("%d", &n);
    int arr[n];
    for(int i=0;i<n;i++){
        scanf("%d", &arr[i]);
    }
    int stack[n],top=-1;
    for(int i=n-1;i>=0;i--){
        int val = arr[i];
        while(top!=-1 && stack[top]<=val){
            top--;
        }
        if(top==-1){
            arr[i] = -1;
        }
        else{
            arr[i] = stack[top];
        }
        stack[++top] = val;
    }
    for(int j=0;j<n;j++){
        printf("%d ", arr[j]);
    }
    return 0;
}
