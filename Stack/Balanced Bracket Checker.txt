#include<stdio.h>
#include<stdlib.h>
int main(){
    int n;
    scanf("%d", &n);
    char s[n+1];
    scanf("%s", s);
    int stack[n],top = -1;
    for(int i=0;i<n;i++){
        char val =(char)s[i];
        if(val=='(' || val=='{' || val=='['){
            stack[++top] = val;
        }
        else{
            if(top==-1){
                printf("INVALID!");
                return 0;
            }
            if(val==')'){
                if(stack[top]=='('){
                    top--;
                }
                else{
                    printf("INVALID!");
                    return 0;
                }
            }
            else if(val=='}'){
                if(stack[top]=='{'){
                    top--;
                }
                else{
                    printf("INVALID!");
                    return 0;
                }
            }
            else{
                if(stack[top]=='['){
                    top--;
                }
                else{
                    printf("INVALID!");
                    return 0;
                }
            }
        }
    }
    if(top==-1){
        printf("VALID!");
    }
    else printf("INVALID!");
}
