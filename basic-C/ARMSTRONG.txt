#include <stdio.h>
#include <math.h>
int main(){
    int n;
    scanf("%d", &n);
    int copy = n;
    int sum=0;
    while(n!=0){
        int val = n%10;
        printf("%d\n", val);
        int s = pow(val,3);
        printf("%d\n", s);
        sum+=s;
        printf("%d\n", sum);
        n = n/10;
        printf("%d\n", n);
    }
    if(sum==copy){
        printf("ARMSTRONG NUMBER");
    }
    else{
        printf("NOT AN ARMSTRONG NUMBER");
    }
    return 0;
}
