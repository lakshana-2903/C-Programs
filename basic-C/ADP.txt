#include <stdio.h> 
#include <math.h> 
int main(){     
    int n;     
    scanf("%d", &n);     
    int sum=1;     
    for(int i=2;i<sqrt(n);i++){         
        if(n%i==0){             
            sum+=i;             
            sum+=(n/i);         
            
        }     
        
    }     
    if(sum==n){         
        printf("PERFECT");     
        
    }     
    else if(sum<n){         
        printf("DEFICIENT");     
        
    }     
    else{         
        printf("ABUNTANT");     
        
    }     
    return 0; 
    
}
