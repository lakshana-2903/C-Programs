//approach 1
#include <stdio.h>
int main(){
	int n;
	scanf("%d", &n);
	int arr[n];
	for(int i=0;i<n;i++){
		scanf("%d", &arr[i]);
	}
	for(int i=0;i<2;i++){
		for(int j=i+1;j<n;j++){
			if(arr[i]<arr[j]){
				int temp = arr[i];
				arr[i] = arr[j];
				arr[j] = temp;
			}
		}
	}
	printf("Second Largest number: %d", arr[1]);
	return 0;
}

#include <stdio.h>
#include <limits.h>
int main(){
	int n;
	scanf("%d", &n);
	int arr[n];
	for(int i=0;i<n;i++){
		scanf("%d", &arr[i]);
	}
	int large = INT_MIN;
	int sec_large = INT_MIN;
	for(int i=0;i<n;i++){
		if(arr[i]>large){
			sec_large = large;
			large = arr[i];
		}
		else if(arr[i]>sec_large && arr[i]!=large){
			sec_large = arr[i];
		}
	}
	printf("SECOND LARGE: %d", sec_large);
	return 0;
}
