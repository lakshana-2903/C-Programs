#include <stdio.h>
#include <math.h>
int main(){
	int n;
	scanf("%d", &n);
	int c=0;
	if(n%10==0){
		for(int i=1;i<=sqrt(n);i++){
			if(n%i==0){
				c+=2;
			}
		}
	}
	if(c==2){
		printf("%d", n);
	}
	while(n!=0){
		int val = n%10;
		int count=0;
		if(val==1){
			count-=2;
		}
		for(int i=1;i<=sqrt(val);i++){
			if(val%i==0){
				count+=2;
			}
		}
		if(count==2){
			printf("%d ", val);
		}
		n = n/10;
	}
	return 0;
}
