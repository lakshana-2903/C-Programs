#include <stdio.h>
int main(){
	int n;
	scanf("%d", &n);
	int arr[n];
	for(int i=0;i<n;i++){
		scanf("%d", &arr[i]);
	}
	for(int i=0;i<n;i++){
		int k=i;
		while(k<n-1 && arr[k]==0){  
			k++;
		}
		if(k==n-1 && arr[k]==0){
			continue;
		}
		int temp = arr[i];
		arr[i] = arr[k];
		arr[k] = temp;	
	}
	for(int i=0;i<n;i++){
		printf("%d ", arr[i]);	
	}
	return 0;
}
